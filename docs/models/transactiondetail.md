# TransactionDetail

## Example Usage

```typescript
import { TransactionDetail } from "ynab-ts/models";
import { RFCDate } from "ynab-ts/types";

let value: TransactionDetail = {
  id: "<id>",
  date: new RFCDate("2023-03-10"),
  amount: 666881,
  cleared: "uncleared",
  approved: false,
  accountId: "455275b2-f2f3-40e3-b478-12e0e9e2c147",
  deleted: false,
  accountName: "<value>",
  subtransactions: [
    {
      id: "<id>",
      transactionId: "<id>",
      amount: 239600,
      deleted: true,
    },
  ],
};
```
